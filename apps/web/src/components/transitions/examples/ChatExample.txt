/**
 * Chat Interface Integration Example
 *
 * This example shows how to integrate PageTransition and AnimatedList
 * into the chat interface for smooth message animations.
 *
 * To use: Apply this pattern to apps/web/src/app/(dashboard)/chat/page.tsx
 */

'use client';

import { useState, useEffect, useRef } from 'react';
import { usePathname } from 'next/navigation';
import {
  PageTransition,
  AnimatedList,
  AnimatedListItem,
  DrawerTransition,
} from '@/components/transitions';
import { Button } from '@/components/ui/button';

// Example message type
interface ChatMessage {
  id: string;
  role: 'user' | 'assistant';
  content: string;
  timestamp: Date;
}

// Example chat bubble component
function ChatBubble({ message }: { message: ChatMessage }) {
  const isUser = message.role === 'user';

  return (
    <div
      className={`flex ${isUser ? 'justify-end' : 'justify-start'} mb-4`}
    >
      <div
        className={`max-w-[80%] rounded-lg p-4 ${
          isUser
            ? 'bg-blue-600 text-white'
            : 'bg-gray-100 dark:bg-gray-800 text-gray-900 dark:text-gray-100'
        }`}
      >
        <p>{message.content}</p>
        <div className="text-xs opacity-70 mt-1">
          {message.timestamp.toLocaleTimeString([], {
            hour: '2-digit',
            minute: '2-digit',
          })}
        </div>
      </div>
    </div>
  );
}

// Example chat input component
function ChatInput({ onSend }: { onSend: (message: string) => void }) {
  const [input, setInput] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (input.trim()) {
      onSend(input);
      setInput('');
    }
  };

  return (
    <form onSubmit={handleSubmit} className="p-4 border-t">
      <div className="flex gap-2">
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Type your message..."
          className="flex-1 p-3 border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
        />
        <Button type="submit">Send</Button>
      </div>
    </form>
  );
}

// Example suggestion chips
function SuggestionChips({ onSelect }: { onSelect: (suggestion: string) => void }) {
  const suggestions = [
    'Show my recent invoices',
    'Create new client',
    'Tax summary',
    'Help with expenses',
  ];

  return (
    <div className="flex flex-wrap gap-2 p-4">
      {suggestions.map((suggestion) => (
        <button
          key={suggestion}
          onClick={() => onSelect(suggestion)}
          className="px-4 py-2 text-sm bg-gray-100 dark:bg-gray-800 rounded-full hover:bg-gray-200 dark:hover:bg-gray-700 transition-colors"
        >
          {suggestion}
        </button>
      ))}
    </div>
  );
}

/**
 * Main Chat Page Component
 */
export function ChatPageExample() {
  const pathname = usePathname();
  const [messages, setMessages] = useState<ChatMessage[]>([
    {
      id: '1',
      role: 'assistant',
      content: 'Hi! I'm your AI business assistant. How can I help you today?',
      timestamp: new Date(),
    },
  ]);
  const [isMenuOpen, setIsMenuOpen] = useState(false);
  const messagesEndRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom when new messages arrive
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  const handleSendMessage = (content: string) => {
    const userMessage: ChatMessage = {
      id: Date.now().toString(),
      role: 'user',
      content,
      timestamp: new Date(),
    };

    setMessages((prev) => [...prev, userMessage]);

    // Simulate AI response
    setTimeout(() => {
      const aiMessage: ChatMessage = {
        id: (Date.now() + 1).toString(),
        role: 'assistant',
        content: 'I received your message: "' + content + '". Let me help you with that.',
        timestamp: new Date(),
      };
      setMessages((prev) => [...prev, aiMessage]);
    }, 1000);
  };

  return (
    <PageTransition pageKey={pathname}>
      <div className="flex flex-col h-screen">
        {/* Chat Header */}
        <div className="p-4 border-b flex items-center justify-between">
          <div>
            <h1 className="text-xl font-bold">Business Assistant</h1>
            <p className="text-sm text-gray-500">Always here to help</p>
          </div>
          <Button
            variant="outline"
            size="sm"
            onClick={() => setIsMenuOpen(true)}
          >
            Menu
          </Button>
        </div>

        {/* Messages Area with Animated List */}
        <div className="flex-1 overflow-y-auto p-4">
          {/* Show suggestions if no user messages yet */}
          {messages.filter((m) => m.role === 'user').length === 0 && (
            <SuggestionChips onSelect={handleSendMessage} />
          )}

          {/* Animated message list */}
          <AnimatedList staggerDelay={0.1}>
            {messages.map((message) => (
              <AnimatedListItem key={message.id}>
                <ChatBubble message={message} />
              </AnimatedListItem>
            ))}
          </AnimatedList>

          <div ref={messagesEndRef} />
        </div>

        {/* Chat Input */}
        <ChatInput onSend={handleSendMessage} />

        {/* Side Menu Drawer */}
        <DrawerTransition
          isOpen={isMenuOpen}
          onClose={() => setIsMenuOpen(false)}
          side="right"
          className="bg-white dark:bg-gray-900 w-80 p-6"
        >
          <h2 className="text-xl font-bold mb-4">Chat Options</h2>
          <ul className="space-y-2">
            <li>
              <button className="w-full text-left p-2 hover:bg-gray-100 dark:hover:bg-gray-800 rounded">
                New Conversation
              </button>
            </li>
            <li>
              <button className="w-full text-left p-2 hover:bg-gray-100 dark:hover:bg-gray-800 rounded">
                View History
              </button>
            </li>
            <li>
              <button className="w-full text-left p-2 hover:bg-gray-100 dark:hover:bg-gray-800 rounded">
                Settings
              </button>
            </li>
          </ul>
        </DrawerTransition>
      </div>
    </PageTransition>
  );
}

/**
 * Alternative: Chat with Message Grouping
 *
 * For grouped messages by date:
 */
import { AnimatedGrid, AnimatedGridItem } from '@/components/transitions';

export function GroupedChatExample() {
  const messagesByDate = {
    Today: [
      /* messages */
    ],
    Yesterday: [
      /* messages */
    ],
  };

  return (
    <div className="space-y-6">
      {Object.entries(messagesByDate).map(([date, msgs]) => (
        <div key={date}>
          <h3 className="text-sm font-semibold text-gray-500 mb-2">{date}</h3>
          <AnimatedList>
            {msgs.map((msg: any) => (
              <AnimatedListItem key={msg.id}>
                <ChatBubble message={msg} />
              </AnimatedListItem>
            ))}
          </AnimatedList>
        </div>
      ))}
    </div>
  );
}

/**
 * Typing Indicator Component
 *
 * Animated typing indicator for when AI is responding
 */
import { motion } from 'framer-motion';

export function TypingIndicator() {
  return (
    <div className="flex items-center gap-2 bg-gray-100 dark:bg-gray-800 rounded-lg p-4 max-w-fit">
      <motion.div
        className="w-2 h-2 bg-gray-400 rounded-full"
        animate={{ y: [0, -8, 0] }}
        transition={{ duration: 0.6, repeat: Infinity, delay: 0 }}
      />
      <motion.div
        className="w-2 h-2 bg-gray-400 rounded-full"
        animate={{ y: [0, -8, 0] }}
        transition={{ duration: 0.6, repeat: Infinity, delay: 0.2 }}
      />
      <motion.div
        className="w-2 h-2 bg-gray-400 rounded-full"
        animate={{ y: [0, -8, 0] }}
        transition={{ duration: 0.6, repeat: Infinity, delay: 0.4 }}
      />
    </div>
  );
}
